19 /*******************************************************************************
20  * TRC_CFG_APPTRACE_TRAX_BLOCKING_MODE
21  * This stream port for streaming over XTensa Trax relies on apptrace that utilize
22  * an internal TRAX RAM buffer read by OpenOCD during execution.
23  *
24  * Possible values:
25  * - APPTRACE_TRAX_MODE_BLOCK_IF_FIFO_FULL
26  * - APPTRACE_TRAX_MODE_NO_BLOCK_SKIP (default)
27  *
28  * Using APPTRACE_TRAX_MODE_BLOCK_IF_FIFO_FULL ensure that you get a
29  * complete and valid trace. This may however cause blocking if your streaming
30  * interface isn't fast enough, which may disturb the real-time behavior.
31  *
32  * We therefore recommend APPTRACE_TRAX_MODE_NO_BLOCK_SKIP. In this mode,
33  * Tracealyzer will report lost events if the transfer is not
34  * fast enough. If there are problems, try increasing the apptrace buffer
35  * in sdkconfig.
36  ******************************************************************************/

38  #define TRC_CFG_APPTRACE_TRAX_BLOCKING_MODE 1
39 #else
